package com.spy.transaction;

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.PreparedStatement;
import java.sql.SQLException;

public class TestTransaction {
	public static void main(String[] args){
		Connection conn = null;
		try {
			//1
			Class.forName("com.mysql.jdbc.Driver");

			String url = "jdbc:mysql://localhost:3306/test";
			String user = "root";
			String password = "123456";
			conn = DriverManager.getConnection(url, user, password);
			//
			conn.setAutoCommit(false);

			String sql1 = "update t_department set description = ? where did = ?";
			PreparedStatement pst1 = conn.prepareStatement(sql1);
			pst1.setObject(1, "");
			pst1.setObject(2, 4);
			int len1 = pst1.executeUpdate();
			System.out.println(len1>0?"":"");
			pst1.close();
			
			String sql2 = "update t_employee set salary = salary + ? where did = ?";
			PreparedStatement pst2 = conn.prepareStatement(sql2);
			pst2.setObject(1, 20000);
			pst2.setObject(2, 4);
			int len2 = pst2.executeUpdate();
			System.out.println(len2>0?"":"");
			pst2.close();
			
			conn.commit();
		}catch (Exception e) {
			try {
				if(conn!=null){
					conn.rollback();
				}
			} catch (SQLException e1) {
				e1.printStackTrace();
			}
		} finally{
			try {
				if(conn!=null){
					//
					conn.setAutoCommit(true);
					//
					conn.close();
				}
			} catch (SQLException e) {
				e.printStackTrace();
			}
		}
	}
}

	private static ThreadLocal<Integer> threadLocal = new ThreadLocal<>();
	
	public static Integer getNumber() {
		// 
		Integer num = threadLocal.get();
		if (num == null) {
			//
			num = rand.nextInt(100);
			// 
			threadLocal.set(num);
		}
		return num;
	}

	public static void removeNumber() {
		// 
		Integer num = threadLocal.get();
		if (num != null) {
			// 
			threadLocal.remove();
		}
	}
}

	 *

		// 
		Connection connection = threadLocal.get();
		if (connection == null) {
			// 
			try {
				connection = dataSource.getConnection();
				// 
				threadLocal.set(connection);
			} catch (SQLException e) {
				e.printStackTrace();
			}
		}
		return connection;
	}

	/**
	 * 
	 * 
	 * @param connection
	 */
	public static void releaseConnection() {
		// 
		Connection connection = threadLocal.get();

			try {
				connection.close();
				// 
				threadLocal.remove();
			} catch (SQLException e) {
				e.printStackTrace();
			}
		}
	}

					//(5)
					field.setAccessible(true);
					
					//(6)
					field.set(obj, rs.getObject(i+1));//i+1
				}
				
				list.add(obj);
